WebSocket Events

Checking for WebSocket support

if(window.WebSocket){ alert("Your browser supports WebSocket"); } else{ alert("Your browser donot support WebSocket"); }

Connecting to server

var client_socket = new WebSocket("ws://www.schoolap.com");

Sending message to server

client_socket.send("My first message");

Connection open event

client_socket.onopen = function(e){ alert("Connection is open") }

Message event

client_socket.onmessage = function(e) { alert(e.data); }

Error event

client_socket.onerror = function(e){ alert("Error"+e); };

Connection close event

client_socket.onclose = function(e){ alert("connection closed"+e); }
